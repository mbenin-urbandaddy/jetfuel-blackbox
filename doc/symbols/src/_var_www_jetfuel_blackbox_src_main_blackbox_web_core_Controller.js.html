1 define('core/Controller', ['$', '_', 'Backbone'], function($, _, Backbone) {
  2     'use strict';

  4     var /**
  5          * @constant
  6          * @private
  7          * @type String
  8          * @fieldOf blackbox.web.core.Controller
  9          * @description Value: "view"
 10          */
 11         ROOT_VIEW_PATH = 'view';
 12 
 13     var /**
 14          * @name currentView
 15          * @private
 16          * @fieldOf blackbox.web.core.Controller
 17          * @type blackbox.web.view.Base
 18          * @description Field to keep track of the current page. A falsy value indicates the initial page load.
 19          */
 20         currentView;
 21 
 22     /**
 23      * @name dispatch
 24      * @methodOf blackbox.web.core.Controller
 25      * @private
 26      * @static
 27      * @param {mixed} ...
 28      * @description
 29      * Dispatches a route pattern to a {@link blackbox.web.view.Base} class.
 30      * <br /><br />
 31      * If <code>currentView</code> is unset (<code>undefined</code>), then we assume that this is the first dispatch event,
 32      * indicating our initial page load. In this case, we <em>DO NOT</em> call the {@link blackbox.web.view.Base#load()} method - the only thing to do
 33      * is to instantiate the view class.
 34      * <br /><br />
 35      * If, however, <code>currentView</code> has been set, then we can assume this is a subsequent page request, so
 36      * we <em>DO</em> call {@link blackbox.web.view.Base#load()} to request whichever resources we need to render the Page.
 37      */
 38     function dispatch() {
 39         var args = arguments;
 40 
 41         require([ROOT_VIEW_PATH + '/' + this], function(View) {
 42             var view = new View(), bind = _.bind;
 43 
 44             if(currentView) {
 45 
 46                 view
 47                     .on('load', bind(load, this, view))
 48                     .on('load:complete', bind(render, this, view))
 49                     .on('render:before', bind(prerender, this, view))
 50                     .load.apply(view, args);
 51 
 52             } else {
 53 
 54                 setCurrentView(view);
 55                 
 56             }
 57         });
 58     }
 59 
 60     /**
 61      * @private
 62      * @static
 63      * @methodOf blackbox.web.core.Controller
 64      * @param {blackbox.web.view.Base} view
 65      * TODO Scroll to top of the page or whatever we gotta do...
 66      */
 67     function load(view) {
 68 
 69     }
 70 
 71     /**
 72      * @private
 73      * @static
 74      * @methodOf blackbox.web.core.Controller
 75      * @param {blackbox.web.view.Base} view
 76      * TODO Finish whatever #load() did...
 77      */
 78     function complete() {
 79 
 80     }
 81 
 82     /**
 83      * @private
 84      * @static
 85      * @methodOf blackbox.web.core.Controller
 86      * @param {blackbox.web.view.Base} view
 87      */
 88     function prerender(view) {
 89         currentView && currentView.destroy();
 90     }
 91 
 92     /**
 93      * @private
 94      * @static
 95      * @methodOf blackbox.web.core.Controller
 96      * @param {blackbox.web.view.Base} view
 97      */
 98     function render(view) {
 99         setCurrentView(view.render());
100         complete();
101     }
102 
103     /**
104      * @private
105      * @static
106      * @methodOf blackbox.web.core.Controller
107      * @param {blackbox.web.view.Base} view
108      */
109     function setCurrentView(view) {
110         currentView = view;
111     }
112 
113     /**
114      * @constructor
115      * @name blackbox.web.core.Controller
116      * @version 2.0
117      */
118     function Controller() {}
119 
120     return {
121         /**
122          * @name createRouter
123          * @methodOf blackbox.web.core.Controller
124          * @static
125          * @param {Object} routes
126          * @returns {Backbone.Router}
127          * @description
128          * Factory for creating Backbone.Router instances with support for our custom routing config.
129          * <br /><br />
130          * Backbone.Router#navigate() has been overridden so that, by default, <code>{ trigger: true }</code>.
131          */
132         createRouter: function(routes) {
133             var Router = {
134 
135                     navigate: function(fragment, options) {
136                         return Backbone.Router.prototype.navigate.call(this, fragment, _.extend({ trigger: true }, options || {}));
137                     }
138 
139                 };
140 
141             _.each(routes, function(view) {
142                 Router[view] = _.bind(dispatch, view);
143             });
144 
145             return new (Backbone.Router.extend(Router))({ routes: routes });
146         },
147 
148         /**
149          * @name startSession
150          * @methodOf blackbox.web.core.Controller
151          * @static
152          * @description Wrapper utility for Backbone.History#start()
153          * @param {Boolean} silent
154          * @returns {Boolean}
155          */
156         startSession: function(silent) {
157             return Backbone.history.start({ pushState: history.pushState, silent: silent });
158         }
159 
160     };
