1 define('view/Base', ['$', '_', 'Backbone', 'Handlebars'], function($, _, Backbone, Handlebars) {
  2     'use strict';
  3 
  4     var /**
  5          * @private
  6          * @constant
  7          * @name DEFAULT_CONTENT
  8          * @type String
  9          * @fieldOf blackbox.web.view.Base.prototype
 10          * @description Value: (empty string)
 11          */
 12         DEFAULT_CONTENT = '';
 13 
 14     var /**
 15          * @name content
 16          * @private
 17          * @type String
 18          * @fieldOf blackbox.web.view.Base.prototype
 19          */
 20         content;
 21 
 22     var /**
 23          * @private
 24          * @name config
 25          * @type Object
 26          * @fieldOf blackbox.web.view.Base.prototype
 27          * @description
 28          * <strong>{String} <code>template</code>:</strong> The default template to load for the view. <em>Defaults to </em><code>"error/404"</code><em>.</em><br />
 29          * <strong>{Boolean} <code>css</code>:</strong> Flag to decide whether to load the CSS for the view. <em>Defaults to </em><code>false</code><em>.</em><br />
 30          * <strong>{Boolean} <code>i18n</code>:</strong> Flag to decide whether to load the i18n bundle for the view. <em>Defaults to </em><code>false</code><em>.</em><br />
 31          * <strong>{Object} <code>data</code>:</strong> The context data to render the view template. <em>Defaults to </em><code>{}</code><em>.</em>
 32          */
 33         config = {
 34             template: 'error/404',
 35             css: false,
 36             i18n: false,
 37             data: {}
 38         };
 39 
 40     /**
 41      * @private
 42      * @methodOf blackbox.web.view.Base.prototype
 43      * @returns {Array}
 44      * @description
 45      * Decides which modules to load based on the current configuration.
 46      */
 47     function getConfiguredModules() {
 48         var templatePath = config.template,
 49             modules = ['resource!templates/' + templatePath + '.html?text'],
 50             base = templatePath.split('/').shift();
 51 
 52         config.i18n && modules.push('resource!nls/' + base + '.js?i18n');
 53         config.css && modules.push('resource!' + base + '.css?css');
 54 
 55         return modules;
 56     }
 57 
 58     /**
 59      * @private
 60      * @param {String} content
 61      * @methodOf blackbox.web.view.Base.prototype
 62      */
 63     function setContent(c) {
 64         c && (content = c);
 65     }
 66 
 67     /**
 68      * @lends blackbox.web.view.Base.prototype
 69      */
 70     return Backbone.View.extend({
 71         /**
 72          * @type {String}
 73          */
 74         el: 'main',
 75 
 76         /**
 77          * @constructs
 78          * @version 2.0
 79          * @param {Object} config
 80          * @see blackbox.web.view.Base#config
 81          * @augments Backbone.View
 82          * @description
 83          * Subscriptions:
 84          * <code>load:before</code>,
 85          * <code>load:complete</code>,
 86          * <code>load</code>,
 87          * <code>render:before</code>,
 88          * <code>render</code>,
 89          * <code>destroy:before</code>,
 90          * <code>destroy</code>
 91          */
 92         initialize: function(cfg) {
 93             this.config(cfg);
 94         },
 95 
 96         /**
 97          * @param {Object} config
 98          * @returns {Object}
 99          */
100         config: function(cfg) {
101             if(typeof cfg == 'object') {
102                 return _.clone(config = $.extend(true, config, cfg || {}));
103             } else {
104                 return _.clone(config[cfg]);
105             }
106         },
107 
108         /**
109          * @returns {blackbox.web.view.Base}
110          * @description
111          * Publishes:
112          * <code>load:before</code>,
113          * <code>load:complete</code>,
114          * <code>load</code>
115          */
116         load: function() {
117             var self = this, trigger = _.bind(this.trigger, this);
118 
119             trigger('load:before');
120 
121             require(getConfiguredModules(), function(template, i18n) {
122                 self.config({ data: { locale: i18n || {} } });
123                 setContent(template);
124                 trigger('load:complete');
125             });
126 
127             return trigger('load');
128         },
129 
130         /**
131          * @param {String} content
132          * @returns {blackbox.web.view.Base}
133          * @see blackbox.web.view.Base#load
134          * @description
135          * Publishes: <code>render:before</code>, <code>render</code>
136          * <br /><br />
137          * Render a template to the view.
138          * <br /><br />
139          * If <code>content</code> is <code>undefined</code>, then this method will
140          * use the stored template {@link blackbox.web.view.Base#content} (which gets set on <code>load:complete</code>),
141          * otherwise it will fallback to {@link blackbox.web.view.Base#DEFAULT_CONTENT}
142          */
143         render: function(content) {
144             this.trigger('render:before');
145             content && setContent(content);
146             this.$el.html(Handlebars.compile(this.getContent())(config.data));
147             return this.trigger('render');
148         },
149 
150         /**
151          * @returns {blackbox.web.view.Base}
152          * @description
153          * Publishes: <code>destroy:before</code>, <code>destroy</code>
154          */
155         destroy: function() {
156             this.trigger('destroy:before').undelegateEvents();
157             this.$el.empty();
158             return this.trigger('destroy');
159         },
160 
161         /**
162          * @returns {String}
163          */
164         getContent: function() {
165             return content || DEFAULT_CONTENT;
166         }
167 
168     });
169     
170 });
