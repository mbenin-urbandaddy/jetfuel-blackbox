1 /**
  2  * @fileOverview Parses directive syntax into an object.
  3  * @author <a href="mailto:promano@urbandaddy.com">Pete Romano</a>
  4  */
  5 define('util/DirectiveParser', function() {
  6     'use strict';
  7 
  8     var /**
  9          * @constant
 10          * @private
 11          * @type {RegExp}
 12          * @fieldOf blackbox.web.util.DirectiveParser
 13          * @description Value: /<(.*)>/
 14          */
 15         DIRECTIVE = /<(.*)>/;
 16 
 17     /**
 18      * @methodOf blackbox.web.util.DirectiveParser
 19      * @private
 20      * @static
 21      * @param {String} directive
 22      * @returns {Array|null}
 23      */
 24     function getDirectiveArray(directive) {
 25         var d = directive.match(/<(.*)>/);
 26         return d && d[1] ? d[1].split(':') : d;
 27     }
 28 
 29     /**
 30      * @name blackbox.web.util.DirectiveParser
 31      * @version 2.0
 32      * @constructor
 33      * @description
 34      * The directive syntax is as follows:
 35      * <pre><<em>method</em>[:<em>arg1</em>][:<em>arg2</em>][<em>...</em>]></pre>
 36      *
 37      *
 38      */
 39     function DirectiveParser() {}
 40 
 41     return {
 42         /**
 43          * @name parse
 44          * @methodOf blackbox.web.util.DirectiveParser
 45          * @static
 46          * @param {String} directive
 47          * @returns {Object|Boolean}
 48          *
 49          * @example
 50          * var app = Application.getInstance(),
 51          *     parsed = DirectiveParser.parse('<execute:Navigator:reload>');
 52          *
 53          * // parsed == { method: 'execute', args: ['Navigator', 'reload'] }
 54          * app[parsed.method].apply(app, parsed.args);
 55          *
 56          * @description
 57          * The method name to invoke is the first segment, followed by the argument list. The context to invoke the
 58          * method is up to you after you parse the directive, OR use the utility method {@link blackbox.web.util.DirectiveParser.invoke}
 59          * to automatically invoke the method with the supplied context.
 60          *
 61          * Returns false if the supplied directive cannot be parsed.
 62          */
 63         parse: function(directive) {
 64             if(!(directive = getDirectiveArray(directive))) {
 65                 return false;
 66             } else {
 67                 return {
 68                     method: directive.shift(),
 69                     args: directive
 70                 };
 71             }
 72         },
 73 
 74         /**
 75          * @name invoke
 76          * @methodOf blackbox.web.util.DirectiveParser
 77          * @static
 78          * @param {String} directive
 79          * @param {Object} context
 80          * @param {int} delay
 81          * @returns {mixed|Boolean|int}
 82          *
 83          * @example
 84          * // Translates to setTimeout(function() { Application.getInstance().execute('Navigator', 'reload'); }, 2500);
 85          * DirectiveParser.invoke('<execute:Navigator:reload>', Application.getInstance(), 2500);
 86          *
 87          * @description
 88          * Returns false if the supplied directive cannot be parsed.<br />
 89          * Returns the return value of the invocation if <pre style="display:inline;">delay</pre> is not specified (the call is synchronous).<br />
 90          * Returns the reference ID from the setTimeout call if <pre style="display:inline;">delay</pre> is specified (the call is asynchronous).
 91          */
 92         invoke: function(directive, context, delay) {
 93             var d = this.parse(directive);
 94 
 95             function invoke() {
 96                 return context[d.method].apply(context, d.args);
 97             }
 98 
 99             if(!d) {
100                 return false;
101             } else if(delay > 0) {
102                 setTimeout(invoke, delay);
103             } else {
104                 return invoke();
105             }
106         }
107 
108     };
109 
110 });
