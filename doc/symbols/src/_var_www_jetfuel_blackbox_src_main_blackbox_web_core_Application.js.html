1 (function() {
  2     'use strict';
  3 
  4     var /**
  5          * @constant
  6          * @private
  7          * @type String
  8          * @fieldOf blackbox.web.core.Application.prototype
  9          * @description Value: "2.0"
 10          */
 11          VERSION = '2.0';
 12 
 13     var /**
 14          * @private
 15          * @name router
 16          * @type Backbone.Router
 17          * @fieldOf blackbox.web.core.Application.prototype
 18          */
 19         router,
 20 
 21         /**
 22          * @private
 23          * @name layout
 24          * @type blackbox.web.view.Layout
 25          * @fieldOf blackbox.web.core.Application.prototype
 26          */
 27         layout,
 28 
 29         /**
 30          * @private
 31          * @name app
 32          * @type blackbox.web.core.Application
 33          * @fieldOf blackbox.web.core.Application.prototype
 34          */
 35         app;
 36 
 37     var /**
 38          * @private
 39          * @name config
 40          * @type {Object}
 41          * @fieldOf blackbox.web.core.Application.prototype
 42          */
 43         config = {
 44             context: window,
 45             routing: {},
 46             settings: {
 47                 debug: false,
 48                 facebook: {},
 49                 google: {
 50                     analytics: {}
 51                 }
 52             }
 53         };
 54 
 55     /**
 56      * @private
 57      * @methodOf blackbox.web.core.Application.prototype
 58      */
 59     function lambda(value) {
 60         return function() { return value; };
 61     }
 62 
 63     /**
 64      * @private
 65      * @methodOf blackbox.web.core.Application.prototype
 66      */
 67     function getApplicationClass() {
 68         return app && app.constructor;
 69     }
 70 
 71     /**
 72      * @private
 73      * @methodOf blackbox.web.core.Application.prototype
 74      */
 75     function domready() {
 76         
 77     }
 78 
 79     /**
 80      * @static
 81      * @methodOf blackbox.web.core.Application
 82      * @returns {blackbox.web.core.Application}
 83      */
 84     function getInstance() {
 85         var Application = this;
 86         return app || (app = new Application());
 87     }
 88 
 89     /**
 90      * @private
 91      * @methodOf blackbox.web.core.Application.prototype
 92      */
 93     function unload() {
 94         layout.destroy();
 95     }
 96 
 97     define('core/Application', ['_', '$', 'Backbone', 'core/Controller', 'view/Layout'], function(_, $, Backbone, Controller, Layout) {
 98         var Application = getApplicationClass(), clone = _.clone;
 99 
100         if(!Application) {
101             /**
102              * @lends blackbox.web.core.Application.prototype
103              * @augments Backbone.Events
104              */
105             (Application = lambda()).prototype = _.extend({
106                 /**
107                  * @constructs
108                  * @version 2.0
109                  * @augments Backbone.Events
110                  * @param {Object} config
111                  * @description
112                  * Subscriptions: <code>domready</code>, <code>unload</code>
113                  * <br />
114                  * Publishes: <code>domready</code>, <code>unload</code>
115                  */
116                 initialize: function(cfg) {
117                     // Set Application#initialize() as a noop, with the same return signature
118                     this.initialize = lambda(this);
119                     // Set application config
120                     this.config(cfg);
121                     // Create and store a new router instance
122                     router = Controller.createRouter(config.routing);
123                     // Start the history session (triggers the initial page view)
124                     Controller.startSession();
125                     // Instantiate the Layout view
126                     layout = new Layout();
127                     // Set handlers for Application domready and unload events
128                     this.on('domready', domready).on('unload', unload);
129                     // Trigger Application domready events on domready event
130                     $(_.bind(this.trigger, this, 'domready'));
131                     // Trigger Application unload events on context onload event
132                     $(config.context).on('unload', _.bind(this.trigger, this, 'unload'));
133                     // Return the Application API
134                     return this;
135                 },
136 
137                 /**
138                  * @param {Object} config
139                  * @returns {Object}
140                  */
141                 config: function(cfg) {
142                     if(typeof cfg == 'object') {
143                         return clone(config = $.extend(true, config, cfg || {}));
144                     } else {
145                         return clone(config[cfg]);
146                     }
147                 },
148 
149                 /**
150                  * @returns {Backbone.Router}
151                  */
152                 getRouter: function() {
153                     return clone(router);
154                 },
155 
156                 /**
157                  * @returns {blackbox.web.view.Layout}
158                  */
159                 getLayout: function() {
160                     return clone(layout);
161                 },
162 
163                 /**
164                  * @function
165                  * @returns {String}
166                  */
167                 getVersion: lambda(VERSION)
168 
169             }, Backbone.Events);
170 
171             Application.getInstance = _.bind(getInstance, Application);
172         }
173 
174         return Application;
175     });

177 })();
