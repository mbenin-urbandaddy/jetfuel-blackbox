1 /**
  2  * @fileOverview Parses a string into URL parts
  3  * @author <a href="mailto:promano@urbandaddy.com">Pete Romano</a>
  4  */
  5 define('util/URLParser', function() {
  6     'use strict';
  7 
  8     var /**
  9          * @constant
 10          * @private
 11          * @type {RegExp}
 12          * @fieldOf blackbox.web.util.URLParser
 13          * @description Value: /^((\w+)\:)?\/\/([^\/\\]+)/
 14          */
 15         X_DOMAIN = /^((\w+)\:)?\/\/([^\/\\]+)/,
 16         /**
 17          * @constant
 18          * @private
 19          * @type {RegExp}
 20          * @fieldOf blackbox.web.util.URLParser
 21          * @description Value: /^(?:([A-Za-z]+:))?(?:\/{2})?([0-9.\-A-Za-z]*)(?::(\d+))?(?:(\/[^?#]*))?(?:\?([^#]*))?(?:#!?(.*))?$/
 22          */
 23         SEGMENTS = /^(?:([A-Za-z]+:))?(?:\/{2})?([0-9.\-A-Za-z]*)(?::(\d+))?(?:(\/[^?#]*))?(?:\?([^#]*))?(?:#!?(.*))?$/,
 24         /**
 25          * @constant
 26          * @private
 27          * @type {RegExp}
 28          * @fieldOf blackbox.web.util.URLParser
 29          * @description Value: /([^&=]+)=?([^&]*)/g
 30          */
 31         QS_PARTS = /([^&=]+)=?([^&]*)/g,
 32         /**
 33          * @constant
 34          * @private
 35          * @type {RegExp}
 36          * @fieldOf blackbox.web.util.URLParser
 37          * @description Value: /\+/g
 38          */
 39         QS_SPACE = /\+/g,
 40         /**
 41          * @private
 42          * @fieldOf blackbox.web.util.URLParser
 43          * @namespace
 44          */
 45         SegmentsEnum = {
 46             /**
 47              * @name blackbox.web.util.URLParser.SegmentsEnum.PROTOCOL
 48              * @constant
 49              * @type {int}
 50              * @description Value: 1
 51              */
 52             PROTOCOL:		1,
 53             /**
 54              * @name blackbox.web.util.URLParser.SegmentsEnum.DOMAIN
 55              * @constant
 56              * @type {int}
 57              * @description Value: 2
 58              */
 59             DOMAIN:			2,
 60             /**
 61              * @name blackbox.web.util.URLParser.SegmentsEnum.PORT
 62              * @constant
 63              * @type {int}
 64              * @description Value: 3
 65              */
 66             PORT:			3,
 67             /**
 68              * @name blackbox.web.util.URLParser.SegmentsEnum.PATH
 69              * @constant
 70              * @type {int}
 71              * @description Value: 4
 72              */
 73             PATH:			4,
 74             /**
 75              * @name blackbox.web.util.URLParser.SegmentsEnum.QUERY_STRING
 76              * @constant
 77              * @type {int}
 78              * @description Value: 5
 79              */
 80             QUERY_STRING:	5,
 81             /**
 82              * @name blackbox.web.util.URLParser.SegmentsEnum.HASH
 83              * @constant
 84              * @type {int}
 85              * @description Value: 6
 86              */
 87             HASH:			6
 88         };
 89 
 90     /**
 91      * @methodOf blackbox.web.util.URLParser
 92      * @private
 93      * @static
 94      * @param {String} url
 95      * @param {int} segment
 96      * @returns {String}
 97      */
 98     function getSegment(url, segment) {
 99         return [].concat(url.match(SEGMENTS) || [])[segment] || '';
100     }
101 
102     /**
103      * @methodOf blackbox.web.util.URLParser
104      * @private
105      * @static
106      * @param {String} url
107      * @returns {Array}
108      */
109     function getSegments(url) {
110         return [].concat(url.match(SEGMENTS) || []);
111     }
112 
113     getSegment.curry = function(segment) {
114         return function(url) { return getSegment(url, segment); };
115     };
116     
117     /**
118      * @name blackbox.web.util.URLParser
119      * @version 2.0
120      * @constructor
121      * @description
122      * Parses a string into URL parts 
123      */
124     function URLParser() {}
125     
126     return {
127         /**
128          * @name parse
129          * @methodOf blackbox.web.util.URLParser
130          * @static
131          * @param {String} url
132          * @returns {Object}
133          */
134         parse: function(url) {
135             var segments = getSegments(url);
136 
137             function getSegment(segment) {
138                 return segments[segment] || '';
139             }
140 
141             return {
142                 port:           getSegment(SegmentsEnum.PORT),
143                 path:           getSegment(SegmentsEnum.PATH),
144                 queryString:    getSegment(SegmentsEnum.QUERY_STRING),
145                 protocol:       getSegment(SegmentsEnum.PROTOCOL),
146                 domain:         getSegment(SegmentsEnum.DOMAIN),
147                 hash:           getSegment(SegmentsEnum.HASH)
148             };
149         },
150 
151         /**
152          * @name getHash
153          * @methodOf blackbox.web.util.URLParser
154          * @static
155          * @param {String} url
156          * @returns {String}
157          */
158         getHash: getSegment.curry(SegmentsEnum.HASH),
159 
160         /**
161          * @name getDomain
162          * @methodOf blackbox.web.util.URLParser
163          * @static
164          * @param {String} url
165          * @returns {String}
166          */
167         getDomain: getSegment.curry(SegmentsEnum.DOMAIN),
168 
169         /**
170          * @name getPort
171          * @methodOf blackbox.web.util.URLParser
172          * @static
173          * @param {String} url
174          * @returns {String}
175          */
176         getPort: getSegment.curry(SegmentsEnum.PORT),
177 
178         /**
179          * @name getPath
180          * @methodOf blackbox.web.util.URLParser
181          * @static
182          * @param {String} url
183          * @returns {String}
184          */
185         getPath: getSegment.curry(SegmentsEnum.PATH),
186 
187         /**
188          * @name getQueryString
189          * @methodOf blackbox.web.util.URLParser
190          * @static
191          * @param {String} url
192          * @returns {String}
193          */
194         getQueryString: getSegment.curry(SegmentsEnum.QUERY_STRING),
195 
196         /**
197          * @name getProtocol
198          * @methodOf blackbox.web.util.URLParser
199          * @static
200          * @param {String} url
201          * @returns {String}
202          */
203         getProtocol: getSegment.curry(SegmentsEnum.PROTOCOL),
204 
205         /**
206          * @name parseQueryString
207          * @methodOf blackbox.web.util.URLParser
208          * @static
209          * @param {String} queryString
210          * @returns {Object}
211          */
212         parseQueryString: function(queryString) {
213             var d = function (s) { return decodeURIComponent(s.replace(QS_SPACE, ' ')); }, p = {}, e;
214 
215             while ((e = QS_PARTS.exec(queryString))) {
216                 p[d(e[1])] = d(e[2]);
217             }
218 
219             return p;
220         },
221 
222         /**
223          * @name isXDomain
224          * @methodOf blackbox.web.util.URLParser
225          * @static
226          * @param {String} url
227          * @returns {Boolean}
228          *
229          * TODO: Add more sophisticated checks against protocol, port, domain, etc.
230          */
231         isXDomain: function(url) {
232             return !!this.getProtocol(url);
233         }
234 
235     };
236 
237 });
